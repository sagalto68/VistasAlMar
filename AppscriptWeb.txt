function getEmailTranslations(lang, type) {
  const defaultLang = 'es';
  const safeLang = EMAIL_TRANSLATIONS[lang] ? lang : defaultLang;
  return EMAIL_TRANSLATIONS[safeLang][type] || EMAIL_TRANSLATIONS[defaultLang][type];
}

function formatEmailText(text, replacements) {
  let result = text;
  for (const [key, value] of Object.entries(replacements)) {
    result = result.replace(new RegExp(`{${key}}`, 'g'), value);
  }
  return result;
}

function getLocalizedDate(date, lang) {
  const locales = {
    es: 'es-ES',
    ca: 'ca-ES', 
    fr: 'fr-FR',
    en: 'en-GB'
  };
  const locale = locales[lang] || 'es-ES';
  return date.toLocaleDateString(locale, {
    weekday: 'long',
    year: 'numeric',
    month: 'long',
    day: 'numeric'
  });
}

function onSheetEdit(e) {
  const sheet = e.range.getSheet();
  if (sheet.getName() !== 'Reservas') return;
  const col = e.range.getColumn(), row = e.range.getRow();
  if (row <= 1) return;

  if (col === 10) {
    const newStatus = e.range.getValue().toString().trim();
    const rowData   = sheet.getRange(row, 1, 1, 14).getValues()[0];
    const booking   = {
      id: rowData[0], nombre: rowData[2], email: rowData[3],
      checkin: rowData[4], checkout: rowData[5], personas: rowData[6],
      telefono: rowData[7], total: rowData[8], estado: rowData[9],
      motivo: rowData[10], calEventId: rowData[11], desglose: rowData[12],
      lang: rowData[13] || 'es'
    };
    if (newStatus === 'Confirmada') {
      sendConfirmationEmail(booking);
      addToCalendar(booking);
      e.range.setBackground('#c6efce').setFontColor('#276221');
    }
    if (newStatus === 'Cancelada') {
      sendCancellationEmail(booking);
      removeFromCalendar(booking);
      e.range.setBackground('#ffc7ce').setFontColor('#9c0006');
    }
  }
}

function getPricing() {
  const sheet = getSS().getSheetByName('Precios');
  const data  = sheet.getDataRange().getValues();
  
  // Fila 1: fechas, Fila 2: precios, Fila 5: limpieza
  const fechasRow   = data[0];  // ['dia', fecha1, fecha2, ...]
  const preciosRow  = data[1];  // ['importe', 60, 60, ...]
  const limpiezaVal = data[4][1]; // Fila 5, columna B
  
  const dailyPrices = {};
  
  for (let col = 1; col < fechasRow.length; col++) {
    if (!fechasRow[col]) continue;
    const fecha = Utilities.formatDate(new Date(fechasRow[col]), 'Europe/Madrid', 'yyyy/MM/dd');
    const precio = parseFloat(preciosRow[col]) || 0;
    dailyPrices[fecha] = precio;
  }
  
  return {
    dailyPrices: dailyPrices,
    limpieza: parseFloat(limpiezaVal) || 80
  };
}

function getBookings() {
  const bookings = [];
  const uniqueRanges = new Set();

  // Fuente 1: Google Calendar
  try {
    const calendar = CalendarApp.getCalendarById(CONFIG.CALENDAR_ID);
    if (calendar) {
      const start = new Date();
      start.setMonth(start.getMonth() - 1);
      const end = new Date();
      end.setFullYear(end.getFullYear() + 2);
      
      const events = calendar.getEvents(start, end);
      Logger.log('Calendar events found: ' + events.length);
      
      events.forEach(ev => {
        const title = ev.getTitle();
        if (title.includes('RESERVADO') || title.includes('🏖️')) {
          const checkin  = Utilities.formatDate(ev.getStartTime(), 'Europe/Madrid', 'yyyy/MM/dd');
          const checkout = Utilities.formatDate(ev.getEndTime(),   'Europe/Madrid', 'yyyy/MM/dd');
          const key = checkin + '|' + checkout;
          if (!uniqueRanges.has(key)) {
            bookings.push({ checkin, checkout, estado: 'Confirmada' });
            uniqueRanges.add(key);
          }
        }
      });
    } else {
      Logger.log('WARNING: Calendar not accessible: ' + CONFIG.CALENDAR_ID);
    }
  } catch(err) {
    Logger.log('Calendar error: ' + err.toString());
  }

  // Fuente 2: Sheet pendientes
  try {
    const sheet = getSS().getSheetByName('Reservas');
    const data  = sheet.getDataRange().getValues();
    for (let i = 1; i < data.length; i++) {
      if (data[i][9] === 'Pendiente' && data[i][0]) {
        const checkin  = Utilities.formatDate(new Date(data[i][4]), 'Europe/Madrid', 'yyyy/MM/dd');
        const checkout = Utilities.formatDate(new Date(data[i][5]), 'Europe/Madrid', 'yyyy/MM/dd');
        const key = checkin + '|' + checkout;
        if (!uniqueRanges.has(key)) {
          bookings.push({ checkin, checkout, estado: 'Pendiente' });
          uniqueRanges.add(key);
        }
      }
    }
  } catch(err) {
    Logger.log('Sheet error: ' + err.toString());
  }

  Logger.log('Total bookings: ' + bookings.length);
  return { bookings };
}

function createBooking(data) {
  const sheet = getSS().getSheetByName('Reservas');
  const id    = 'RES-' + new Date().getTime().toString().slice(-6);
  
  sheet.appendRow([
    id, new Date(), data.nombre, data.email,
    new Date(data.checkin), new Date(data.checkout),
    data.personas, data.telefono || '',
    data.montoTotal,       // I: total
    'Pendiente',           // J: estado
    '',                    // K: motivo cancelación
    '',                    // L: ID evento calendario
    data.desglose || '',   // M: desglose textual
    data.lang || 'es'      // N: idioma
  ]);
  
  sendOwnerNotification(data, id);
  sendPendingConfirmation(data, id);
  return { success: true, bookingId: id };
}

function updateBookingStatus(id, newStatus, motivo) {
  const ss = getSS();
  const sheet = ss.getSheetByName('Reservas');
  const data  = sheet.getDataRange().getValues();

  for (let i = 1; i < data.length; i++) {
    if (data[i][0] === id) {
      const row = data[i];
      const booking = {
        id: row[0],
        fechaSolicitud: row[1],
        nombre: row[2],
        email: row[3],
        checkin: row[4],
        checkout: row[5],
        personas: row[6],
        telefono: row[7],
        total: row[8],
        estado: newStatus,
        motivo: motivo || row[10],
        calEventId: row[11],
        desglose: row[12],
        lang: row[13] || 'es'
      };

      // Actualizamos estado y motivo en la hoja
      const estadoCell = sheet.getRange(i + 1, 10); // J
      estadoCell.setValue(newStatus);
      if (motivo) {
        sheet.getRange(i + 1, 11).setValue(motivo); // K
      }

      // Colores igual que en onSheetEdit
      if (newStatus === 'Confirmada') {
        estadoCell.setBackground('#c6efce').setFontColor('#276221');
      } else if (newStatus === 'Cancelada') {
        estadoCell.setBackground('#ffc7ce').setFontColor('#9c0006');
      } else {
        estadoCell.setBackground(null).setFontColor(null);
      }

      // Acciones según nuevo estado
      if (newStatus === 'Confirmada') {
        sendConfirmationEmail(booking);
        addToCalendar(booking);
      } else if (newStatus === 'Cancelada') {
        sendCancellationEmail(booking);
        removeFromCalendar(booking);
      }

      return { success: true, id: id, estado: newStatus };
    }
  }

  return { success: false, error: 'Reserva no encontrada: ' + id };
}

function confirmBooking(id) {
  return updateBookingStatus(id, 'Confirmada');
}

function cancelBooking(id, motivo) {
  return updateBookingStatus(id, 'Cancelada', motivo || '');
}

function addToCalendar(b) {
  try {
    const calendar = CalendarApp.getCalendarById(CONFIG.CALENDAR_ID);
    if (!calendar) {
      Logger.log('ERROR: Calendario no accesible: ' + CONFIG.CALENDAR_ID);
      MailApp.sendEmail({
        to: CONFIG.OWNER_EMAIL,
        subject: '⚠️ Error: calendario no accesible',
        body: 'No se pudo acceder al calendario ' + CONFIG.CALENDAR_ID + 
              '. Asegúrate de:\n1. Ejecutar desde aptopda121@gmail.com, o\n2. Compartir el calendario con sagalto68@gmail.com'
      });
      return;
    }
    
    const checkin  = new Date(b.checkin);
    const checkout = new Date(b.checkout);
    const nights   = Math.round((checkout - checkin) / (1000*60*60*24));

    const event = calendar.createAllDayEvent(
      '🏖️ RESERVADO — ' + b.nombre + ' (' + nights + 'n)',
      checkin, checkout,
      {
        description:
          'Huésped: '  + b.nombre + '\n' +
          'Email: '    + b.email + '\n' +
          'Teléfono: ' + (b.telefono || '—') + '\n' +
          'Personas: ' + b.personas + '\n' +
          'Total: '    + b.total + '€\n' +
          'ID: '       + b.id,
        location: CONFIG.LOCATION,
        color: CalendarApp.EventColor.RED
      }
    );

    const sheet = getSS().getSheetByName('Reservas');
    const data  = sheet.getDataRange().getValues();
    for (let i = 1; i < data.length; i++) {
      if (data[i][0] === b.id) {
        sheet.getRange(i + 1, 12).setValue(event.getId());
        break;
      }
    }
    
    Logger.log('Evento creado: ' + event.getId());
  } catch(err) {
    Logger.log('Error calendario: ' + err.toString());
    MailApp.sendEmail({
      to: CONFIG.OWNER_EMAIL,
      subject: '⚠️ Error al crear evento',
      body: 'No se pudo crear evento para reserva ' + b.id + '.\nError: ' + err.toString()
    });
  }
}

function removeFromCalendar(b) {
  try {
    if (!b.calEventId) return;
    const calendar = CalendarApp.getCalendarById(CONFIG.CALENDAR_ID);
    if (!calendar) return;
    const event = calendar.getEventById(b.calEventId);
    if (event) {
      event.setTitle('❌ CANCELADO — ' + b.nombre);
      event.setColor(CalendarApp.EventColor.GRAY);
    }
  } catch(err) {
    Logger.log('Error al cancelar: ' + err.toString());
  }
}

function trackVisit(params) {
  const sheet = getSS().getSheetByName('Visitas');
  const now = new Date();
  sheet.appendRow([now, Utilities.formatDate(now,'Europe/Madrid','yyyy/MM/dd'), Utilities.formatDate(now,'Europe/Madrid','HH:mm'), params.page||'/', params.ref||'directo', params.device||'unknown']);
  return { tracked: true };
}

function getStats() {
  const ss = getSS();
  const visData = ss.getSheetByName('Visitas').getDataRange().getValues();
  const resData = ss.getSheetByName('Reservas').getDataRange().getValues();
  const today = Utilities.formatDate(new Date(),'Europe/Madrid','yyyy/MM/dd'), month = today.substring(0,7);
  let vToday=0, vMonth=0, total=0, pending=0, revenue=0;
  for(let i=1;i<visData.length;i++){const d=visData[i][1];if(d===today)vToday++;if(d&&d.toString().startsWith(month))vMonth++;}
  for(let i=1;i<resData.length;i++){if(resData[i][0]){total++;if(resData[i][9]==='Pendiente')pending++;if(resData[i][9]!=='Cancelada')revenue+=parseFloat(resData[i][8])||0;}}
  return {visitsToday:vToday,visitsMonth:vMonth,totalVisits:visData.length-1,totalBookings:total,pendingBookings:pending,revenue};
}

function doGet(e) {
  Logger.log('doGet params: ' + JSON.stringify(e && e.parameter));
  const action = e.parameter.action || '';
  const callback = e.parameter.callback || '';
  let result;

  try {
    if (action === 'getBookings') {
      result = getBookings();
    } else if (action === 'getPricing') {
      result = getPricing();
    } else if (action === 'trackVisit') {
      result = trackVisit(e.parameter);
    } else if (action === 'getStats' && e.parameter.secret === CONFIG.ADMIN_SECRET) {
      result = getStats();
    } else if (action === 'confirmBooking' && e.parameter.id && e.parameter.secret === CONFIG.ADMIN_SECRET) {
      const r = confirmBooking(e.parameter.id);
      const msg = r.success ? '✅ Reserva ' + e.parameter.id + ' confirmada correctamente.' : '❌ ' + r.error;
      return HtmlService.createHtmlOutput(
        '<html><body style="font-family:Arial,sans-serif;padding:24px;"><h2>' +
        msg +
        '</h2><p>Ya puedes cerrar esta ventana.</p></body></html>'
      );
    } else if (action === 'cancelBooking' && e.parameter.id && e.parameter.secret === CONFIG.ADMIN_SECRET) {
      const motivo = e.parameter.reason || '';
      const r = cancelBooking(e.parameter.id, motivo);
      const msg = r.success ? '✅ Reserva ' + e.parameter.id + ' cancelada.' : '❌ ' + r.error;
      return HtmlService.createHtmlOutput(
        '<html><body style="font-family:Arial,sans-serif;padding:24px;"><h2>' +
        msg +
        '</h2><p>Ya puedes cerrar esta ventana.</p></body></html>'
      );
    } else if (action === 'getPendingBookings' && e.parameter.secret === CONFIG.ADMIN_SECRET) {
      result = getPendingBookings();
    } else {
      result = { error: 'Unknown action' };
    }
  } catch (err) {
    result = { error: err.toString() };
  }

  const json = JSON.stringify(result);
  if (callback) {
    return ContentService
      .createTextOutput(callback + '(' + json + ')')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  return ContentService
    .createTextOutput(json)
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  let result;
  try {
    const data = JSON.parse(e.postData.contents);
    if (data.action === 'createBooking') {
      result = createBooking(data);
    } else {
      result = { error: 'Unknown action' };
    }
  } catch (err) {
    result = { error: err.toString() };
  }

  return ContentService
    .createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

function showPendingBookings() {
  const sheet = getSS().getSheetByName('Reservas');
  if (!sheet) return;

  const range = sheet.getDataRange();
  let filter = sheet.getFilter();
  if (!filter) {
    filter = range.createFilter();
  }

  // Columna J (Estado) -> índice 10
  const criteria = SpreadsheetApp.newFilterCriteria()
    .setHiddenValues(['Confirmada', 'Cancelada', '']) // mostramos solo "Pendiente"
    .build();

  filter.setColumnFilterCriteria(10, criteria);
}

function showAllBookings() {
  const sheet = getSS().getSheetByName('Reservas');
  if (!sheet) return;

  const filter = sheet.getFilter();
  if (filter) {
    filter.remove();
  }
}

function getPendingBookings() {
  var ss = getSS();
  var sheet = ss.getSheetByName('Reservas');
  if (!sheet) return { pending: [] };
  var data = sheet.getDataRange().getValues();
  if (!data || data.length < 2) return { pending: [] };
  var tz = 'Europe/Madrid';
  var pending = [];
  for (var i = 1; i < data.length; i++) {
    var row = data[i];
    var estado = (row[9] && row[9].toString().trim()) ? row[9].toString().trim() : '';
    if (estado !== 'Pendiente') continue;
    var checkin = row[4];
    var checkout = row[5];
    if (checkin && (checkin instanceof Date)) checkin = Utilities.formatDate(checkin, tz, 'yyyy/MM/dd');
    else if (checkin) checkin = checkin.toString();
    if (checkout && (checkout instanceof Date)) checkout = Utilities.formatDate(checkout, tz, 'yyyy/MM/dd');
    else if (checkout) checkout = checkout.toString();
    pending.push({
      id: (row[0] && row[0].toString()) ? row[0].toString() : '',
      nombre: (row[2] && row[2].toString()) ? row[2].toString() : '',
      email: (row[3] && row[3].toString()) ? row[3].toString() : '',
      checkin: checkin || '',
      checkout: checkout || '',
      total: (row[8] != null && row[8] !== '') ? Number(row[8]) : ''
    });
  }
  return { pending: pending };
}
